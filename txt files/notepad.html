<!-- JavaScript Animation -->
<script>
  const farmerStory = document.querySelector('.farmer-story');
  const sectionTitle = document.querySelector('.section-title');
  const farmerImg = document.querySelector('.farmer-img');
  const storyText = document.querySelector('.story');
  const thoughtsText = document.querySelector('.thoughts');

  // Animate elements on scroll
  const animateOnScroll = () => {
    const elements = [sectionTitle, farmerImg, storyText, thoughtsText];
    elements.forEach((element, index) => {
      setTimeout(() => {
        element.style.opacity = 1;
        element.style.transform = 'translateY(0)';
      }, 500 + (index * 300));
    });
  };

  // Check if element is in viewport
  const isInViewport = (element) => {
    const rect = element.getBoundingClientRect();
    return (
      rect.top <= (window.innerHeight || document.documentElement.clientHeight) &&
      rect.bottom >= 0
    );
  };

  // Add event listener to window
  window.addEventListener('scroll', () => {
    if (isInViewport(farmerStory)) {
      animateOnScroll();
      window.removeEventListener('scroll', animateOnScroll);
    }
  });
</script>
